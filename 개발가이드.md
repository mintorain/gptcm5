# 사이버펑크 네온 랜딩페이지 — 재현 개발 가이드

> 이 문서는 GPT 콘텐츠 메이커 5기 랜딩페이지와 **동일한 구조의 사이트를 재현**하기 위한 기술 가이드입니다.
> AI 코딩 도구(Claude, Cursor 등)에 이 문서를 제공하면 동일한 구조의 랜딩페이지를 생성할 수 있습니다.

---

## 1. 아키텍처 개요

- **단일 HTML 파일** (index.html)에 모든 CSS, JS 인라인 포함
- 빌드 도구 없음, 서버 없음 — 정적 사이트
- Vercel 또는 GitHub Pages로 즉시 배포 가능

```
기술 스택:
HTML5 + TailwindCSS CDN + Vanilla JS
Google Fonts (Noto Sans KR, Orbitron)
Material Symbols Outlined (아이콘)
Google Apps Script (폼 데이터 수집)
```

### 4.2 카운트다운 타이머

```javascript
var DEADLINE = new Date('2026-02-19T00:00:00+09:00');
function updateCountdown() {
    var diff = DEADLINE - new Date();
    if (diff <= 0) {
        // 마감 처리: CTA 비활성화, 텍스트 변경, Floating CTA 숨김
        return;
    }
    var d = Math.floor(diff / 864e5);
    var h = Math.floor((diff % 864e5) / 36e5);
    var m = Math.floor((diff % 36e5) / 6e4);
    var s = Math.floor((diff % 6e4) / 1e3);
    var t = 'D-' + d + '일 ' + String(h).padStart(2,'0') + ':' +
            String(m).padStart(2,'0') + ':' + String(s).padStart(2,'0') + ' 남음';
    document.getElementById('countdownText').textContent = '마감까지 ' + t;
}
updateCountdown();
setInterval(updateCountdown, 1000);
```

### 4.3 Google Sheets 연동 (Hidden Iframe — CORS 우회)

**Google Apps Script 코드 (구글시트 측):**
```javascript
function doPost(e) {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    sheet.appendRow([
        new Date(),
        e.parameter.name,
        e.parameter.phone,
        e.parameter.email,
        e.parameter.source
    ]);
    return ContentService.createTextOutput("OK");
}
function doGet(e) { return doPost(e); }
```

**프론트엔드 전송:**
```html
<!-- body 끝에 hidden iframe 추가 -->
<iframe name="hidden_iframe" id="hidden_iframe" style="display:none;"></iframe>
```

```javascript
var SCRIPT_URL = "https://script.google.com/macros/s/YOUR_DEPLOY_ID/exec";

function submitToSheet(data) {
    var f = document.createElement('form');
    f.method = 'POST';
    f.action = SCRIPT_URL;
    f.target = 'hidden_iframe';     // ← CORS 우회 핵심
    f.acceptCharset = 'UTF-8';      // ← 한글 깨짐 방지
    Object.keys(data).forEach(function(k) {
        var i = document.createElement('input');
        i.type = 'hidden'; i.name = k; i.value = data[k];
        f.appendChild(i);
    });
    document.body.appendChild(f);
    f.submit();
    document.body.removeChild(f);
}
```

> **주의**: `fetch`로 전송하면 CORS 오류 발생. 반드시 hidden iframe 방식 사용.

```javascript
var tracks = [
    { t: '트랙1 이름', s: '트랙1.mp3' },

audio.play().then(function() {
    playing = true;
}).catch(function() {
    document.addEventListener('click', function autoPlay() {
        if (!playing) { audio.play(); playing = true; }
    }, { once: true });

// 트랙 순환
audio.addEventListener('ended', function() {
    ci = (ci + 1) % tracks.length;
    audio.src = tracks[ci].s;

// YouTube iframe 클릭 시 BGM 정지 (window blur 감지)
window.addEventListener('blur', function() {

            audio.pause();
```

### 4.8 애니메이션 모음

```javascript
// 숫자 카운터 (IntersectionObserver + requestAnimationFrame)
function animateCounter(el) {
    var target = parseInt(el.getAttribute('data-target'), 10), dur = 1500, start = performance.now();
    function easeOut(t) { return 1 - Math.pow(1 - t, 3); }
    function tick(now) {
        var p = Math.min((now - start) / dur, 1);
        el.textContent = Math.round(easeOut(p) * target);
        if (p < 1) requestAnimationFrame(tick);
    }
    requestAnimationFrame(tick);
}
new IntersectionObserver(function(entries) {
    entries.forEach(function(e) { if (e.isIntersecting) { animateCounter(e.target); } });
}, { threshold: 0.3 }).observe(/* 각 카운터 요소 */);

// 스크롤 리빌 (섹션 등장 효과)
new IntersectionObserver(function(entries) {
    entries.forEach(function(e) {
        if (e.isIntersecting) {
            e.target.style.opacity = '1';
            e.target.style.transform = 'translateY(0)';
        }
    });
}, { threshold: 0.1 });

// 컨페티 (Canvas API 자체 구현)
```

## 8. 주의사항 및 트러블슈팅

| 문제 | 원인 | 해결 |
|------|------|------|
| YouTube 임베드 오류 | `youtube.com` 도메인 사용 | `youtube-nocookie.com` 사용 |
| Google Sheets 기록 안됨 | `fetch` CORS 차단 | hidden iframe 방식 사용 |
| 한글 깨짐 | 인코딩 미지정 | `form.acceptCharset = 'UTF-8'` |
| BGM 자동재생 안됨 | 브라우저 자동재생 정책 | `catch`로 첫 클릭 시 재생 대기 |
| Vercel Next.js 빌드 오류 | framework 자동 감지 | `vercel.json`에 `"framework": null` |
| 라이트모드에서 글자 안 보임 | CSS 변수 미적용 | `html.light` 오버라이드 CSS 필수 |
| 영상 클릭해도 BGM 안 꺼짐 | iframe 크로스오리진 | `window.blur` 이벤트로 감지 |
| 중복 폼 제출 | 상태 관리 없음 | `localStorage` 플래그 + 버튼 비활성화 |
